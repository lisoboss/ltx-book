### 11.1.2 FP8 量化

```bash
# 显存节省约 40%，速度提升约 30-50%

# FP8 Cast (通用方案)
PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True \
python -m ltx_pipelines.ti2vid_two_stages \
    --quantization fp8-cast \
    ...

# FP8 Scaled MM (Hopper GPU + TensorRT-LLM)
PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True \
python -m ltx_pipelines.ti2vid_two_stages \
    --quantization fp8-scaled-mm \
    ...
```

### 11.1.3 torch.compile

```bash
# 编译 transformer blocks 加速推理

# 默认编译

# reduce-overhead 模式 (启用 CUDA graphs, 更快但多占显存)
python -m ltx_pipelines.ti2vid_two_stages --compile mode=reduce-overhead ...

# max-autotune (首次慢, 后续快)
python -m ltx_pipelines.ti2vid_two_stages --compile mode=max-autotune fullgraph=true ...
```


### 11.3.3 种子选择策略

```python
import random

def find_best_seed(pipeline, prompt, images, num_tries=5):
    """
    快速尝试多个 seed，选最好的
    蒸馏模型 2 分钟/次 × 5 = 10 分钟找到好 seed
    """
    best_video = None
    best_score = -float("inf")

    for i in range(num_tries):
        seed = random.randint(0, 2**32 - 1)
        video, _ = pipeline(
            prompt=prompt, seed=seed,
            images=images, ...
        )
        # 用 CLIP score / 美学评分 / 人工判断
        score = aesthetic_score(video)
        if score > best_score:
            best_score = score
            best_video = video

    return best_video
```

## 11.4 多 LoRA 融合技巧

### 11.4.1 LoRA 强度调度

```python
# 不同镜头使用不同的 LoRA 强度
# 例如: 特写镜头提高角色 LoRA 强度，远景降低

def adaptive_lora_strength(shot_type: str) -> float:
    return {
        "extreme_closeup": 1.0,   # 大特写 → 最强角色约束
        "closeup": 0.9,
        "medium": 0.8,            # 中景 → 标准
        "wide": 0.6,              # 远景 → 降低 (给环境更多自由)
        "extreme_wide": 0.4,      # 大远景 → 最低 (角色很小)
    }.get(shot_type, 0.8)
```

## 11.6 批量处理与并行

### 11.6.1 多 GPU 并行

```python
# 方案 1: 多进程，每个 GPU 一个进程
import subprocess
import multiprocessing as mp

def generate_on_gpu(gpu_id, shots):
    env = {**os.environ, "CUDA_VISIBLE_DEVICES": str(gpu_id)}
    for shot in shots:
        subprocess.run([
            "python", "-m", "ltx_pipelines.ti2vid_two_stages",
            "--prompt", shot["prompt"],
            "--output-path", shot["output"],
            ...
        ], env=env)

# 将镜头分给不同 GPU
shots_per_gpu = len(all_shots) // num_gpus
processes = []
for gpu_id in range(num_gpus):
    start = gpu_id * shots_per_gpu
    end = start + shots_per_gpu
    p = mp.Process(target=generate_on_gpu, args=(gpu_id, all_shots[start:end]))
    p.start()
    processes.append(p)

for p in processes:
    p.join()
```

### 11.6.2 预热与缓存

```python
# 首次运行会编译 CUDA kernels，较慢
# 用一个小视频预热 GPU

def warmup_gpu(pipeline):
    """预热: 跑一个最小的推理"""
    pipeline(
        prompt="warmup",
        seed=0, height=256, width=256, num_frames=9,  # 最小尺寸
        num_inference_steps=2,  # 最少步数
        ...
    )
    torch.cuda.synchronize()
    print("GPU warmed up!")

# 预热后，实际生成速度提升 20-50%（主要是 CUDA kernel 编译缓存）
```
