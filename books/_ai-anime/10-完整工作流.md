### 10.3.1 场景生成脚本

```python
#!/usr/bin/env python3
"""
AI 动漫场景批量生成脚本
从分镜脚本自动生成所有场景视频
"""

import json
import os
from pathlib import Path
from ltx_core.loader import LTXV_LORA_COMFY_RENAMING_MAP, LoraPathStrengthAndSDOps
from ltx_core.components.guiders import MultiModalGuiderParams
from ltx_pipelines.ti2vid_two_stages import TI2VidTwoStagesPipeline
from ltx_pipelines.keyframe_interpolation import KeyframeInterpolationPipeline
from ltx_pipelines.utils.args import ImageConditioningInput
from ltx_pipelines.utils.media_io import encode_video

class AnimeSceneGenerator:
    def __init__(self, project_root: str):
        self.project_root = Path(project_root)
        self.device = torch.device("cuda")

        # 加载角色和风格 LoRA
        self.loras = [
            LoraPathStrengthAndSDOps(
                str(self.project_root / "loras/char_yozakura.safetensors"),
                0.8, LTXV_LORA_COMFY_RENAMING_MAP,
            ),
            LoraPathStrengthAndSDOps(
                str(self.project_root / "loras/style_anime.safetensors"),
                0.5, LTXV_LORA_COMFY_RENAMING_MAP,
            ),
        ]

        self.distilled_lora = [
            LoraPathStrengthAndSDOps(
                "models/ltx-2.3-22b-distilled-lora-384-1.1.safetensors",
                0.6, LTXV_LORA_COMFY_RENAMING_MAP,
            ),
        ]

        # 双 Pipeline: I2V + 关键帧插值
        self.i2v_pipeline = TI2VidTwoStagesPipeline(
            checkpoint_path="models/ltx-2.3-22b-dev.safetensors",
            distilled_lora=self.distilled_lora,
            spatial_upsampler_path="models/ltx-2.3-spatial-upscaler-x2-1.1.safetensors",
            gemma_root="models/gemma-3-12b",
            loras=self.loras,
        )

        self.kf_pipeline = KeyframeInterpolationPipeline(
            checkpoint_path="models/ltx-2.3-22b-dev.safetensors",
            distilled_lora=self.distilled_lora,
            spatial_upsampler_path="models/ltx-2.3-spatial-upscaler-x2-1.1.safetensors",
            gemma_root="models/gemma-3-12b",
            loras=self.loras,
        )

        self.video_guider = MultiModalGuiderParams(
            cfg_scale=3.0, stg_scale=1.0, rescale_scale=0.7,
            modality_scale=1.0, stg_blocks=[28],
        )

    def generate_from_storyboard(self, storyboard_path: str):
        """从分镜脚本批量生成"""
        with open(storyboard_path) as f:
            storyboard = json.load(f)

        for shot in storyboard["shots"]:
            print(f"Generating {shot['id']}: {shot['description'][:50]}...")

            output_path = self.project_root / "outputs" / shot["id"]
            output_path.mkdir(parents=True, exist_ok=True)

            # 根据关键帧数量选择 pipeline
            keyframes = self._load_keyframes(shot)

            if len(keyframes) == 1:
                # 单关键帧 → I2V
                video, audio = self.i2v_pipeline(
                    prompt=shot["prompt"],
                    negative_prompt=shot.get("negative_prompt",
                        "worst quality, low quality, blurry, deformed"),
                    seed=shot.get("seed", 42),
                    height=shot.get("height", 512),
                    width=shot.get("width", 768),
                    num_frames=self._duration_to_frames(shot["duration_sec"]),
                    frame_rate=24.0,
                    num_inference_steps=shot.get("steps", 40),
                    video_guider_params=self.video_guider,
                    audio_guider_params=self.audio_guider,
                    images=[ImageConditioningInput(
                        keyframes[0], frame_idx=0, strength=1.0, crf=33
                    )],
                )
            else:
                # 多关键帧 → 关键帧插值
                kf_inputs = [
                    ImageConditioningInput(kf["path"], kf["frame_idx"],
                                           kf.get("strength", 1.0), 33)
                    for kf in keyframes
                ]
                video, audio = self.kf_pipeline(
                    prompt=shot["prompt"],
                    negative_prompt=shot.get("negative_prompt", "..."),
                    seed=shot.get("seed", 42),
                    height=shot.get("height", 512),
                    width=shot.get("width", 768),
                    num_frames=self._duration_to_frames(shot["duration_sec"]),
                    frame_rate=24.0,
                    num_inference_steps=shot.get("steps", 40),
                    video_guider_params=self.video_guider,
                    audio_guider_params=self.audio_guider,
                    images=kf_inputs,
                )

            # 保存
            encode_video(
                video=video, fps=24.0, audio=audio,
                output_path=str(output_path / f"{shot['id']}_take01.mp4"),
            )

            # 如果启用，生成多个 take 供选择
            if shot.get("multi_take", False):
                for take in range(2, shot.get("num_takes", 3) + 1):
                    # 换 seed 重试
                    ...

    def _duration_to_frames(self, seconds: float) -> int:
        """将时长转换为满足 8k+1 的帧数"""
        target = int(seconds * 24)
        # 找到最近的满足 8k+1 的帧数
        return ((target - 1) // 8) * 8 + 1

    def _load_keyframes(self, shot: dict) -> list:
        """加载关键帧"""
        kf_dir = self.project_root / "keyframes" / shot["id"]
        return sorted(kf_dir.glob("*.jpg"))
```

## 10.5 实战：3 分钟动漫短片制作预算

```
3 分钟 = 180 秒 @ 24fps = ~4320 帧

制作估算 (使用 RTX 4090):

Phase 1 - 前期 (1 周):
├── 剧本/分镜: 2 天
├── 角色设计 + LoRA 训练: 3 天 (含 12h 训练)
└── 关键帧制作 (SD/Flux): 2 天

Phase 2 - 中期 (3-5 天):
├── 场景生成 (15 个镜头): 1-2 天
│   └── 每个镜头 ~5-10min (蒸馏模型预览) + ~20-30min (完整模型最终)
├── 版本迭代/重试: 1-2 天
└── IC-LoRA 风格统一: 1 天

Phase 3 - 后期 (2-3 天):
├── Retake 修复: 1-2 天
├── LipDub 配音: 1 天
└── 剪辑+调色+导出: 1 天

总计: 约 2 周 (单人, RTX 4090)
对比传统动画: 3 分钟 ≈ 2-4 个月 (5-10 人团队)
```

## 10.6 批量自动化框架

```python
# 完整自动化管线
class AnimeProductionPipeline:
    """
    从分镜到成片的自动化管线
    """

    def run(self, project_config: dict):
        # 1. 验证项目结构
        self.validate_project()

        # 2. 顺序生成各场景
        for shot in project_config["shots"]:
            result = self.generate_shot(shot)

            # 3. 自动质量检查
            qc = self.quality_check(result["video_path"])

            # 4. 不合格则自动修复
            retry = 0
            while not qc["pass"] and retry < 3:
                fix = self.auto_fix(result["video_path"], qc["issues"])
                result = fix
                qc = self.quality_check(result["video_path"])
                retry += 1

            if not qc["pass"]:
                print(f"WARNING: {shot['id']} needs manual review")

        # 5. 场景串联
        self.concatenate_scenes()

        # 6. 最终导出
        self.export_final()

    def generate_shot(self, shot):
        """单个镜头生成 + 自动选择最佳 take"""
        best_score = -float("inf")
        best_result = None

        for take in range(shot.get("num_takes", 1)):
            result = self._run_generation(shot, take)
            score = self._score_take(result)
            if score > best_score:
                best_score = score
                best_result = result

        return best_result
```
